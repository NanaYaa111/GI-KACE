    const email = prompt("Enter your email:");
    const password = prompt("Enter your password:");

    // Check if any prompt was cancelled (which returns null)
    if (name === null,  email === null , password === null) {
        alert("Registration cancelled.");
        return null;
    }
    if (password.length >= 6) {
        return {
            name: name,
            email: email,
            password: password
        };
    } else {
            alert("Password too short! Must be at least 6 characters.");
        return null;
    }
}



// 9: Temperature Converter
function convertTemp(value, unit) {
    let result;

    if (unit === "C") {
        result = (value * 9/5) + 32;
    } else if (unit === "F") {
        result = (value - 32) * 5/9;
    } else {
        return "Invalid unit specified. Use 'C' or 'F'.";
    }
    return parseFloat(result.toFixed(1));
}
console.log(`25 C to F: ${convertTemp(25, "C")}`);  
console.log(`98.6 F to C: ${convertTemp(98.6, "F")}`);    


// 10: Simple Banking System
let balance = 1000;

/**
 * @param {number} amount 
 * @returns {number | string} 
 */
function deposit(amount) {
    if (amount <= 0 || typeof amount !== 'number') {
        return "Deposit amount must be a positive number."; // Not allowed negative deposits
    }
    balance += amount;
    balance = parseFloat(balance.toFixed(2));
    return balance;
}

/**
 * @param {number} amount 
 * @returns {number | string} 
 */
function withdraw(amount) {
    if (amount <= 0 || typeof amount !== 'number') {
        return "Withdrawal amount must be a positive number."; [span_24](start_span)// Not allowed negative withdrawals[span_24](end_span)
    }
    if (balance >= amount) {
        balance -= amount;
        balance = parseFloat(balance.toFixed(2));
        return balance;
    } else {                        
        return "Insufficient funds";
    }
}
console.log("Initial Balance: " + balance);
console.log(`Deposit 250.50: ${deposit(250.50)}`);  
console.log(`Withdraw 300: ${withdraw(300)}`);
